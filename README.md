<script>
document.addEventListener("DOMContentLoaded", function() {
    const rawData = document.getElementById('quiz-data').textContent;
    
    // Мәтінді сұрақтар блогына бөлу
    const blocks = rawData.split('\\/');
    const quizData = [];

    blocks.forEach(block => {
        let cleanBlock = block.replace('/\\', '').trim();
        if (!cleanBlock) return;
        
        let lines = cleanBlock.split('\n').map(l => l.trim()).filter(l => l.length > 0);
        let questionText = "";
        let options = [];
        
        lines.forEach(line => {
            if (line.startsWith('+')) {
                options.push({ text: line.substring(1).trim(), isCorrect: true });
            } else if (line.startsWith('-')) {
                options.push({ text: line.substring(1).trim(), isCorrect: false });
            } else {
                questionText += line + " ";
            }
        });
        
        if (questionText && options.length > 0) {
            quizData.push({ question: questionText.trim(), options: options });
        }
    });

    // Жауап нұсқаларын араластыру (Shuffle)
    quizData.forEach(q => {
        for (let i = q.options.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [q.options[i], q.options[j]] = [q.options[j], q.options[i]];
        }
    });

    // Тестті HTML-ге шығару
    const container = document.getElementById('quiz-content');
    quizData.forEach((q, qIndex) => {
        const qDiv = document.createElement('div');
        qDiv.className = 'question-block';
        
        const qTitle = document.createElement('div');
        qTitle.className = 'question-title';
        qTitle.textContent = `${qIndex + 1}. ${q.question}`;
        qDiv.appendChild(qTitle);
        
        const ul = document.createElement('ul');
        ul.className = 'options';
        
        q.options.forEach((opt, optIndex) => {
            const li = document.createElement('li');
            const label = document.createElement('label');
            
            const radio = document.createElement('input');
            radio.type = 'radio';
            radio.name = `question_${qIndex}`;
            radio.value = opt.isCorrect;
            
            // Жауапты таңдаған кездегі оқиға
            radio.addEventListener('change', function() {
                const allLis = ul.querySelectorAll('li');
                
                // Барлық нұсқаларды бұғаттау
                allLis.forEach(item => {
                    item.classList.add('disabled');
                    item.querySelector('input').disabled = true;
                    
                    // Дұрыс жауапты жасылмен көрсету
                    if (item.querySelector('input').value === 'true') {
                        item.classList.add('correct');
                    }
                });
                
                // Егер таңдалған жауап қате болса, қызылмен көрсету
                if (this.value === 'false') {
                    li.classList.add('incorrect');
                }
            });
            
            label.appendChild(radio);
            label.appendChild(document.createTextNode(opt.text));
            li.appendChild(label);
            ul.appendChild(li);
        });
        
        qDiv.appendChild(ul);
        container.appendChild(qDiv);
    });

    // Нәтижені есептеу батырмасы
    document.getElementById('submit-btn').addEventListener('click', () => {
        const selected = document.querySelectorAll('input[type="radio"]:checked');
        let score = 0;
        
        selected.forEach(radio => {
            if (radio.value === 'true') {
                score++;
            }
        });
        
        const resultDiv = document.getElementById('result');
        resultDiv.textContent = `Сіздің нәтижеңіз: ${score} / ${quizData.length}`;
        resultDiv.style.display = 'block';
        
        document.getElementById('submit-btn').disabled = true;
        
        // Нәтижеге қарай экранды жылжыту
        window.scrollTo({ top: document.body.scrollHeight, behavior: 'smooth' });
    });
});
</script>
